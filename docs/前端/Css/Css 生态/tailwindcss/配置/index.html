配置

/** @type {import('tailwindcss').Config} */
export default {
  // content 属性用于告诉 Tailwind 哪些文件需要扫描以生成相应的 CSS。
  content: [],

  // theme 属性用于自定义 Tailwind 的设计系统，包括颜色、字体、断点、间距等。
  theme: {
    extend: {
      // 颜色配置，你可以自定义或扩展这些颜色。
      colors: {
        "custom-blue": "#1fb6ff",
      },
      // 定义间距和尺寸，包括 padding、margin、width、height 等。
      spacing: {
        128: "32rem",
      },
      // 定义字体家族，可以是系统字体或自定义字体。
      fontFamily: {
        sans: ["Graphik", "sans-serif"],
        serif: ["Merriweather", "serif"],
      },
      // 定义响应式设计的断点。
      screens: {
        xl: "1200px",
        "2xl": "1536px",
      },
    },
  },

  // plugins 属性用于引入 Tailwind 插件，扩展其功能。
  plugins: [],
};
